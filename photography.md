<script>
(function() {
  var gallery = document.querySelector('.gallery');
  var items = Array.prototype.slice.call(gallery.children);
  var mq1 = window.matchMedia('(max-width: 30rem)');
  var mq2 = window.matchMedia('(max-width: 48rem)');

  function layout() {
    var cols = mq1.matches ? 1 : mq2.matches ? 2 : 3;
    gallery.innerHTML = '';
    var columns = [];
    for (var c = 0; c < cols; c++) {
      var col = document.createElement('div');
      col.className = 'gallery-col';
      gallery.appendChild(col);
      columns.push(col);
    }
    for (var i = 0; i < items.length; i++) {
      columns[i % cols].appendChild(items[i]);
    }
  }

  layout();
  mq1.addEventListener('change', layout);
  mq2.addEventListener('change', layout);
})();
</script>
